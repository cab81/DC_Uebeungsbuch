\chapter{Welpen \& Junghunde}
\label{chap:heim:welpen}

\section{Grundübungen}

% ── Übung: Sitz ───────────────────────────────────────────────────────────────
\begin{uebung}{Sitz}{welpen}{Grundkommando}{2}
  \uebungsbeschreibung{%
    Halte einen Leckerli-Happen knapp über die Nase des Hundes und führe ihn langsam nach hinten über den Kopf. Der Hund wird automatisch in die Sitz-Position gehen, um dem Leckerli zu folgen.\\[6pt]
    \textbf{Schritte:}
    \begin{enumerate}[leftmargin=*, itemsep=2pt]
      \item Leckerli in geschlossener Faust über Hundenase halten.
      \item Faust langsam nach hinten-oben führen.
      \item Sobald Hund sitzt: markieren (Clicker / „Ja!") und belohnen.
      \item Kommando erst einführen, wenn Verhalten zuverlässig gezeigt wird.
    \end{enumerate}
  }
  \uebungsvariationen{%
    \begin{itemize}[leftmargin=*, itemsep=4pt]
      \item \textbf{Mit Distanz:} Sitz aus zunehmend größerer Entfernung üben.
      \item \textbf{Mit Ablenkung:} Übung in belebter Umgebung wiederholen.
      \item \textbf{Dauer steigern:} Hund lernt, in der Sitz-Position zu verweilen, bis er freigegeben wird.
    \end{itemize}
  }
  \uebungstrainer{%
    \begin{itemize}[leftmargin=*, itemsep=4pt]
      \item Darauf achten, dass der Hund nicht springt – Faust nicht zu hoch halten.
      \item Timing des Markers ist entscheidend: exakt im Moment des Hinsetzens.
      \item Bei Welpen kurze Trainingseinheiten (max. 3–5 Minuten).
    \end{itemize}
  }
\end{uebung}
