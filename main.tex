\newenvironment{uebung}[4]{%
  % #1 = Titel, #2 = altersgruppe, #3 = Kategorie, #4 = Schwierigkeit
  \def\UebTitel{#1}%
  \def\UebAlter{#2}%
  \def\UebKategorie{#3}%
  \def\UebSchwierigkeit{#4}%
  \def\UebBild{}%
  \def\UebBeschreibung{}%
  \def\UebVariationen{}%
  \def\UebTrainer{}%
}{%
  \clearpage
  % ── Linke Seite (gerade Seitenzahl) ──────────────────────────────────────
  \ifodd\value{page}\hbox{}\clearpage\fi
  % Seite 1: Titel + Metadaten + Bild + Beschreibung
  % Header: Titelzeile + Metaraster – beide exakt \textwidth breit
  {%
    \setlength{\fboxsep}{0pt}%
    \noindent\colorbox{primary}{%
      \parbox[c][1.05cm][c]{\textwidth}{%
        \hspace{8pt}{\color{white}\huge\bfseries\UebTitel}%
      }%
    }%
  }\\[-2pt]%
  \noindent{%
    \setlength{\tabcolsep}{6pt}%
    \arrayrulecolor{white!30!primary}%
    \begin{tabularx}{\textwidth}{%
      >{\hsize=1.20\hsize\centering\arraybackslash\color{white}\footnotesize}X|
      >{\hsize=0.80\hsize\centering\arraybackslash\color{white}\footnotesize}X%
    }
      \cellcolor{primary!80!black}\rule{0pt}{0.55cm}\faTag~\UebKategorie &
      \cellcolor{primary!80!black}\schwierigkeit{\UebSchwierigkeit}\\[3pt]
    \end{tabularx}%
  }%

  \vspace{8pt}

  % Bild
  \ifthenelse{\equal{\UebBild}{}}{%
    % Platzhalterbild
    \begin{tcolorbox}[colback=gray!10, colframe=gray!30, arc=4pt, boxrule=0.8pt, height=5.5cm, valign=center, halign=center]
      \textcolor{gray}{\faImage[regular]~\large Bild einfügen}
    \end{tcolorbox}%
  }{%
    \begin{center}
      \includegraphics[width=\textwidth, height=5.5cm, keepaspectratio]{\UebBild}
    \end{center}%
  }

  \vspace{10pt}

  % Beschreibung
  \begin{tcolorbox}[
    title={\faBook~Beschreibung},
    colback=infobg,
    colframe=primary,
    fonttitle=\bfseries\color{white},
    coltitle=white,
    attach boxed title to top left={yshift=-2mm, xshift=4mm},
    boxed title style={colback=primary, arc=3pt},
    arc=4pt, boxrule=0.8pt,
    breakable
  ]
    \UebBeschreibung
  \end{tcolorbox}

  \clearpage

  % ── Rechte Seite (ungerade Seitenzahl) ───────────────────────────────────
  % Seitenüberschrift
  \begin{tcolorbox}[
    colback=primary,
    colframe=primary,
    arc=0pt,
    boxrule=0pt,
    left=6pt, right=6pt, top=5pt, bottom=5pt,
    width=\textwidth,
    sharp corners
  ]
    {\color{white}\large\bfseries\faUserTie\quad Trainer-Informationen}
  \end{tcolorbox}

  \vspace{8pt}

  % Variationen
  \begin{tcolorbox}[
    colback=trainerbg,
    colframe=primary,
    arc=4pt, boxrule=0.8pt,
    breakable,
    top=6pt, bottom=6pt, left=6pt, right=6pt
  ]
    \noindent{\color{primary}\bfseries\faRandom\quad Variationen}\par\vspace{4pt}
    \UebVariationen
  \end{tcolorbox}

  \vspace{10pt}

  % Trainer-Hinweise
  \begin{tcolorbox}[
    colback=trainerbg,
    colframe=primary,
    arc=4pt, boxrule=0.8pt,
    breakable,
    top=6pt, bottom=6pt, left=6pt, right=6pt
  ]
    \noindent{\color{primary}\bfseries\faExclamationCircle\quad Hinweise für Trainer}\par\vspace{4pt}
    \UebTrainer
  \end{tcolorbox}

  \clearpage
}
